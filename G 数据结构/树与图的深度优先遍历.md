## [P2625 - 树的重心 - ZJHUOJ](http://172.20.8.83/problem.php?id=2625)
```cpp
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
//const int inf=0x3f3f3f3f;
const int N=1e5+5;
//树的重心
int  n,h[N],e[N<<1],ne[N<<1],st[N],idx,ans=N;
void add(int a,int b){//链式前向星//可以看做链表的插入(idx看做指针)
    e[idx]=b;//指向的节点，即树的子节点
    ne[idx]=h[a];//下一个指向上个(未插入前的)头节点
    h[a]=idx++;//指向插入的头并自加
}
int dfs(int u){
    int sum=1,res=0;
    st[u]=1;
    for(int i=h[u];i!=-1;i=ne[i]){
        int v=e[i];
        if(!st[v]){
        int m=dfs(v);
        res=max(res,m);
        sum+=m;
        }
    }
    res=max(res,n-sum);
    ans=min(ans,res);
    return sum;
}
void solve(){
    memset(h,-1,sizeof h);
    int a,b; cin>>n;
    for(int i=1;i<n;++i){
        cin>>a>>b;
        add(a,b); add(b,a);
    }
    dfs(1);
    cout<<ans<<endl;
}
signed main(){
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    int T=1;
   // cin>>T;
    while(T--)solve();
    return 0;
}
```
