```cpp
#include<bits/stdc++.h>
using namespace std; 
#define endl '\n'
void solve(){
string s; int ans=0,n;
cin>>n>>s;
s='0'+s;
char ch=s[0];
for(int i=0;i<=n;++i){
    if(ch!=s[i])++ans;
    ch=s[i];
}
if(ans>=3)cout<<ans-2+n<<endl;
else if(ans==2)cout<<ans-1+n<<endl;
else cout<<ans+n<<endl;
}
signed main(){
    ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
    int T=1;
    cin>>T;
    while(T--)solve();
    return 0;
}
```
